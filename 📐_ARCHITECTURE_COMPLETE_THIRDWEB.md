```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";

/**
 * @title PaieCashStablecoin
 * @dev Contrat pour g√©rer les stablecoins (USDC, cUSD)
 */
contract PaieCashStablecoin is ERC20 {
    address public owner;
    
    constructor() ERC20("PaieCash USD", "pcUSD") {
        owner = msg.sender;
    }
    
    /**
     * Mint de nouveaux tokens (recharge on-ramp)
     */
    function mint(address to, uint256 amount) external {
        require(msg.sender == owner, "Only owner can mint");
        _mint(to, amount);
    }
    
    /**
     * Transfert de stablecoin
     */
    function sendStablecoin(address to, uint256 amount) external returns (bool) {
        return transfer(to, amount);
    }
    
    /**
     * Obtenir le solde
     */
    function getBalance(address account) external view returns (uint256) {
        return balanceOf(account);
    }
}
```

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

/**
 * @title ClubMultiSigWallet
 * @dev Portefeuille multi-signatures pour les clubs
 * N√©cessite 3/5 signatures pour les transferts importants
 */
contract ClubMultiSigWallet {
    address[] public signers;
    uint256 public requiredSignatures;
    
    mapping(uint256 => Transaction) public transactions;
    mapping(uint256 => mapping(address => bool)) public confirmations;
    uint256 public transactionCount;
    
    struct Transaction {
        address to;
        uint256 amount;
        bytes data;
        bool executed;
        uint256 confirmationCount;
    }
    
    event TransactionSubmitted(uint256 indexed txId, address indexed to, uint256 amount);
    event TransactionConfirmed(uint256 indexed txId, address indexed signer);
    event TransactionExecuted(uint256 indexed txId);
    
    constructor(address[] memory _signers, uint256 _requiredSignatures) {
        require(_signers.length >= _requiredSignatures, "Invalid config");
        
        signers = _signers;
        requiredSignatures = _requiredSignatures;
    }
    
    /**
     * Soumettre une transaction
     */
    function submitTransaction(address to, uint256 amount, bytes memory data) 
        external 
        returns (uint256) 
    {
        require(isSigner(msg.sender), "Not a signer");
        
        uint256 txId = transactionCount++;
        transactions[txId] = Transaction({
            to: to,
            amount: amount,
            data: data,
            executed: false,
            confirmationCount: 0
        });
        
        emit TransactionSubmitted(txId, to, amount);
        
        // Auto-confirm
        confirmTransaction(txId);
        
        return txId;
    }
    
    /**
     * Confirmer une transaction
     */
    function confirmTransaction(uint256 txId) public {
        require(isSigner(msg.sender), "Not a signer");
        require(!confirmations[txId][msg.sender], "Already confirmed");
        
        confirmations[txId][msg.sender] = true;
        transactions[txId].confirmationCount++;
        
        emit TransactionConfirmed(txId, msg.sender);
        
        // Auto-execute si assez de confirmations
        if (transactions[txId].confirmationCount >= requiredSignatures) {
            executeTransaction(txId);
        }
    }
    
    /**
     * Ex√©cuter une transaction
     */
    function executeTransaction(uint256 txId) internal {
        Transaction storage txn = transactions[txId];
        
        require(!txn.executed, "Already executed");
        require(txn.confirmationCount >= requiredSignatures, "Not enough confirmations");
        
        txn.executed = true;
        
        (bool success, ) = txn.to.call{value: txn.amount}(txn.data);
        require(success, "Transaction failed");
        
        emit TransactionExecuted(txId);
    }
    
    /**
     * V√©rifier si une adresse est un signataire
     */
    function isSigner(address account) public view returns (bool) {
        for (uint256 i = 0; i < signers.length; i++) {
            if (signers[i] == account) {
                return true;
            }
        }
        return false;
    }
}
```

```typescript
import { ThirdwebSDK } from '@thirdweb-dev/sdk';

  process.env.THIRDWEB_SECRET_KEY!,

   * Cr√©er un wallet In-App
   */

      // Cr√©er le wallet via Thirdweb

      const address = await wallet.getAddress();
      
      // Enregistrer dans la DB

        email,
        phone,
        walletAddress: address,
        createdAt: new Date()
      });
      
      // Mapping contact ‚Üí wallet

        contact: email || phone,
        walletAddress: address
      });
      
      // Bonus de bienvenue
      await this.creditWelcomeBonus(address);
      
      return { success: true, address };
      
    } catch (error) {
      console.error('Erreur cr√©ation wallet:', error);
      throw error;
    }
  }
  
  /**
   * Envoyer des stablecoins par contact
   */
  async sendStablecoin(fromAddress: string, contact: string, amount: number) {
    try {
      // Rechercher le wallet du destinataire
      const recipient = await db.contactMapping.findOne({ contact });
      
      if (!recipient) {
        throw new Error('Contact introuvable');
      }
      
      // R√©cup√©rer le contrat stablecoin
      const contract = await sdk.getContract(process.env.USDC_CONTRACT_ADDRESS!);
      
      // Effectuer le transfert
      const tx = await contract.erc20.transfer(recipient.walletAddress, amount);
      
      // Enregistrer la transaction
      await db.transactions.create({
        from: fromAddress,
        to: recipient.walletAddress,
        amount,
        currency: 'USDC',
        txHash: tx.receipt.transactionHash,
        timestamp: new Date()
      });
      
      return {
        success: true,
        txHash: tx.receipt.transactionHash,
        recipient: recipient.walletAddress
      };
      
    } catch (error) {
      console.error('Erreur transfert:', error);
      throw error;
    }
  }
  
  /**
   * Mint un NFT billet
   */
  async mintTicketNFT(to: string, eventId: string, price: number) {
    try {
      // R√©cup√©rer le contrat NFT

        tokenId: tx.id,
        txHash: tx.receipt.transactionHash
      };
      
    } catch (error) {

      throw error;
```
